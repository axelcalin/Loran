85  	/**
86  	 * Loads the map defined by nextMap, if isLoad is set to true.
87  	 * Also adds two lives when going back to the "menu" and addLife is true.
88  	 */
89  	public synchronized void loadMap() {
90  		if(isLoad){
91  			if(nextMap.equalsIgnoreCase("me") && this.addLife){
92  				DAOLife daoLife;
93  				try{
94  					daoLife = new DAOLife(DBConnection.getInstance().getConnection());
95  					daoLife.addLife();
96  				} catch(SQLException e){
97  					e.printStackTrace();
98  				}
99  			}
100 			try {
101 //				this.curMap = this.nextMap;
102 				this.isLoad = false;
103 				final DAOModel daoModel = new DAOModel(DBConnection.getInstance().getConnection());
104 				this.setMap(daoModel.loadMap(nextMap, dynamicElements));
105 			} catch (final SQLException e) {
106 				e.printStackTrace();
107 			}
108 		}
109 	}

183 	public void tick(){
184 		this.loadMap();
185 		Iterator<IElement> dynobj = dynamicElements.iterator();
186 		while(dynobj.hasNext()){
187 				dynobj.next().animate();
188 		}
189 		this.doKill();
190 		this.changed();
191 	}

237 	/**
238 	 * Removes the objects stored in targets from the map.
239 	 */
240 	public void doKill(){
241 		Iterator<IElement> targets = this.killTargets.iterator();
242 		Iterator<IElement> list = this.getDynamicObject().iterator();
243 		while(targets.hasNext()){
244 			IElement target = targets.next();
245 			while(list.hasNext()){
246 				if(target.getClass() == list.next().getClass()){
247 					list.remove();
248 				}
249 			}
250 			list = this.getDynamicObject().iterator();
251 			this.map[((IMobile)target).getY()][((IMobile)target).getX()] = new White();
252 		}
253 		this.killTargets.clear();
254 	}

284 			daoLife.removeLife();

286 				this.saveScore();
