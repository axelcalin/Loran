79  	@Override
80  	protected void paintComponent(final Graphics graphics) {
81  		graphics.clearRect(0, 0, 1500,1000);
82  		int i = 0;
83  		int j = 0;
84  		Iterator<java.util.List<IElement>> n = this.getViewFrame().getModel().getMap().iterator();
85  		while(n.hasNext()){
86  			Iterator<IElement> k = n.next().iterator();
87  			while(k.hasNext()){
88  				graphics.drawImage(k.next().getSprite().getImage(),j*82,i*82, 75, 75,this);
89  				j++;
90  			}
91  			i++;
92  			j = 0;
93  		}
94  		Iterator<IElement> dynamics = this.getViewFrame().getModel().getDynamicObject().iterator();
95  		while(dynamics.hasNext()){
96  			IElement components = dynamics.next();
97  			components.animate();
98  		}
99  	}
